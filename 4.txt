#include <stdio.h>
#include <semaphore.h>

int main()
{
    int ch, phil;

    sem_t chopstick[5];
    sem_t room;

    for(int i = 0; i < 5; i++)
        sem_init(&chopstick[i], 0, 1);

    sem_init(&room, 0, 4);

    while(1)
    {
        printf("\n1.Eat");
        printf("\n2.Finish");
        printf("\n3.Exit");
        printf("\nEnter choice: ");
        scanf("%d", &ch);

        if(ch == 1)
        {
            printf("Enter philosopher number (0-4): ");
            scanf("%d", &phil);

            sem_wait(&room);

            sem_wait(&chopstick[phil]);
            sem_wait(&chopstick[(phil + 1) % 5]);

            printf("Philosopher %d is eating\n", phil);
        }

        else if(ch == 2)
        {
            printf("Enter philosopher number (0-4): ");
            scanf("%d", &phil);

            sem_post(&chopstick[phil]);
            sem_post(&chopstick[(phil + 1) % 5]);

            sem_post(&room);

            printf("Philosopher %d finished eating\n", phil);
        }

        else
        {
            break;
        }
    }

    return 0;
}
