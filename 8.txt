#include <stdio.h>

int main()
{
    int pages[20], frames[10];
    int n, f, ch;

    printf("Enter number of pages: ");
    scanf("%d", &n);

    printf("Enter pages:\n");

    for(int i = 0; i < n; i++)
        scanf("%d", &pages[i]);

    printf("Enter number of frames: ");
    scanf("%d", &f);

    printf("\n1.FIFO\n2.Optimal\n");
    printf("Enter choice: ");
    scanf("%d", &ch);

    for(int i = 0; i < f; i++)
        frames[i] = -1;

    int faults = 0, index = 0;

    for(int i = 0; i < n; i++)
    {
        int found = 0;

        for(int j = 0; j < f; j++)
        {
            if(frames[j] == pages[i])
            {
                found = 1;
                break;
            }
        }

        if(found == 0)
        {
            faults++;

            if(ch == 1)
            {
                frames[index] = pages[i];
                index = (index + 1) % f;
            }

            else
            {
                int pos = 0, far = -1;

                for(int j = 0; j < f; j++)
                {
                    int k;

                    for(k = i + 1; k < n; k++)
                    {
                        if(frames[j] == pages[k])
                            break;
                    }

                    if(k > far)
                    {
                        far = k;
                        pos = j;
                    }
                }

                frames[pos] = pages[i];
            }
        }
    }

    printf("Page Faults = %d\n", faults);

    return 0;
}
