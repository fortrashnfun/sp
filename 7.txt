#include <stdio.h>

int main()
{
    int b, p;

    printf("Enter number of blocks: ");
    scanf("%d", &b);

    int block[b], temp[b];

    printf("Enter block sizes:\n");

    for(int i = 0; i < b; i++)
    {
        scanf("%d", &block[i]);
        temp[i] = block[i];
    }

    printf("Enter number of processes: ");
    scanf("%d", &p);

    int process[p];

    printf("Enter process sizes:\n");

    for(int i = 0; i < p; i++)
        scanf("%d", &process[i]);

    int ch;

    printf("\n1.First Fit\n2.Best Fit\n3.Worst Fit\n");
    printf("Enter choice: ");
    scanf("%d", &ch);

    for(int i = 0; i < p; i++)
    {
        int index = -1;

        for(int j = 0; j < b; j++)
        {
            if(temp[j] >= process[i])
            {
                if(ch == 1)
                {
                    index = j;
                    break;
                }

                else if(ch == 2)
                {
                    if(index == -1 || temp[j] < temp[index])
                        index = j;
                }

                else if(ch == 3)
                {
                    if(index == -1 || temp[j] > temp[index])
                        index = j;
                }
            }
        }

        if(index != -1)
        {
            printf("Process %d allocated to Block %d\n", i + 1, index + 1);

            temp[index] -= process[i];
        }
        else
        {
            printf("Process %d Not Allocated\n", i + 1);
        }
    }

    return 0;
}
