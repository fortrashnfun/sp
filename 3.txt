#include <stdio.h>
#include <semaphore.h>

int readcount = 0;

int main()
{
    int ch;
    sem_t mutex, wrt;

    sem_init(&mutex, 0, 1);
    sem_init(&wrt, 0, 1);

    while(1)
    {
        printf("\n1.Reader Enter\n2.Reader Exit\n3.Writer Write\n4.Exit\n");
        printf("Enter choice: ");
        scanf("%d", &ch);

        if(ch == 1)
        {
            sem_wait(&mutex);

            readcount++;

            if(readcount == 1)
                sem_wait(&wrt);

            sem_post(&mutex);

            printf("Reader is reading\n");
        }

        else if(ch == 2)
        {
            sem_wait(&mutex);

            if(readcount > 0)
            {
                readcount--;

                if(readcount == 0)
                    sem_post(&wrt);

                printf("Reader exited\n");
            }

            sem_post(&mutex);
        }

        else if(ch == 3)
        {
            sem_wait(&wrt);

            printf("Writer is writing\n");

            sem_post(&wrt);
        }

        else
        {
            break;
        }
    }

    return 0;
}
